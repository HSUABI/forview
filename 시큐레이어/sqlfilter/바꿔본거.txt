SELECT DISTINCT a, b, c FROM table_name;
코드거친후 : ['SELECT DISTINCT FROM', 'column',['a','b','c'], 'TABLE', 'table_name']

SELECT a, b, c FROM table_name WHERE name="홍길동";
코드거친후 : ['SELECT FROM', 'column',['a','b','c'], 'TABLE', 'table_name','WHERE',['홍길동'] ] 

SELECT a, b, c FROM table_name WHERE name="홍길동" AND age=20;
코드거친후 : ['SELECT FROM', 'column',['a','b','c'], 'TABLE', 'table_name','WHERE AND',["홍길동",20] ]

SELECT a, b, c FROM table_name WHERE name IN ('홍길동','이영희','김철수');
코드거친후 : ['SELECT FROM', 'column',['a','b','c'], 'TABLE', 'table_name','WHERE IN',['홍길동','이영희','김철수'] ]

SELECT a, b, c FROM table_name WHERE age BETWEEN 20 AND 30;
코드거친후 : ['SELECT FROM', 'column',['a','b','c'], 'TABLE', 'table_name','WHERE BETWEEN AND',[20,30] ]

SELECT a, b, c FROM table_name WHERE name LIKE '홍%';
코드거친후 : ['SELECT FROM', 'column',['a','b','c'], 'TABLE', 'table_name','WHERE LIKE',["홍%"] ]

SELECT a, b, c FROM table_name WHERE age=20 ORDER BY name DESC, score ASC;
코드거친후 : ['SELECT FROM', 'column',['a','b','c'], 'TABLE', 'table_name','WHERE',[20],"ORDER BY",["name","DESC","score","ASC"] ] //ORDERBY는 어떻게 해야할지 몰라서 우선 DESC 랑 ASC를 리스트에 넣었습니다.
